Easing

이동하는 애니메이션입니다.

이동하기

tweenMax를 이용하여 이동하는 애니메이션입니다.

문법(Syntax)

TweenMax.to("target", duration, vars);

이동하기

버튼을 클릭하면 오른쪽으로 이동하는 애니메이션입니다.

JQUERY
$(".start").on("click",function(){
    $(".box1").animate({ left: "80%" }, 2000, "swing");
    $(".box2").animate({ left: "80%" }, 2000, "easeInQuad");
    $(".box3").animate({ left: "80%" }, 2000, "easeOutQuint");
    $(".box4").animate({ left: "80%" }, 2000, "easeInOutBack");
    $(".box5").animate({ left: "80%" }, 2000, "easeOutBounce");
    $(".box6").animate({ left: "80%" }, 2000, "easeOutElastic");
    $(".box7").animate({ left: "80%" }, 2000, "easeInCirc");
    $(".box8").animate({ left: "80%" }, 2000, "easeOutExpo");
});
GSAP
 $(".start").on("click",function(){
    TweenMax.to(".box1",2, {left:"90%"});
    TweenMax.to(".box2",2, {left:"90%", ease:Power0.easeOut});
    TweenMax.to(".box3",2, {left:"90%", ease:Power1.easeOut});
    TweenMax.to(".box4",2, {left:"90%", ease:Power2.easeOut});
    TweenMax.to(".box5",2, {left:"90%", ease:Power3.easeOut});
    TweenMax.to(".box6",2, {left:"90%", ease:Power4.easeOut});
    TweenMax.to(".box7",2, {left:"90%", ease:Bounce.easeOut});
    TweenMax.to(".box8",2, {left:"90%", ease:SlowMo.easeOut});
});
JAVASCRIPT
const startBtn = document.querySelector('.start');
const resetBtn = document.querySelector('.reset');

startBtn.addEventListener('click', moveStart);
resetBtn.addEventListener('click', moveReset);

function moveStart(){
    TweenMax.to(".box1",2, {left:"90%"});
    TweenMax.to(".box2",2, {left:"90%", ease:Power0.easeOut});
    TweenMax.to(".box3",2, {left:"90%", ease:Power1.easeOut});
    TweenMax.to(".box4",2, {left:"90%", ease:Power2.easeOut});
    TweenMax.to(".box5",2, {left:"90%", ease:Power3.easeOut});
    TweenMax.to(".box6",2, {left:"90%", ease:Power4.easeOut});
    TweenMax.to(".box7",2, {left:"90%", ease:Bounce.easeOut});
    TweenMax.to(".box8",2, {left:"90%", ease:SlowMo.easeOut});
}
function moveReset(){
    TweenMax.to([".box1",".box2",".box3",".box4",".box5",".box6",".box7",".box8"], 0, { left: "0", rotationY: "0deg" });
}
