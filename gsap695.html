Repeat

이동하는 애니메이션입니다.

이동하기

tweenMax를 이용하여 이동하는 애니메이션입니다.

문법(Syntax)

TweenMax.to("target", duration, vars);

이동하기

버튼을 클릭하면 오른쪽으로 이동하는 애니메이션입니다.

Sample1
box
리셋 클릭하면 오른쪽으로 움직입니다.
jquery
$(".choice1 .off1").on("click",function(e){
    e.preventDefault();
    TweenMax.to(".box1_1", 0, {left:"0%"});
});
$(".choice1 .btn1").on("click",function(e){
    e.preventDefault();
    TweenMax.to(".box1_1", 1, {left:"80%"});
});
Javascript
var btn1 = document.querySelector('.btn1');
var btn0 = document.querySelector('.off1');

btn1.addEventListener('click', moveStart);
btn0.addEventListener('click', moveReset);

function moveStart(){
    TweenMax.to('.box1_1', 1, {left: "80%"});
}
function moveReset(){
    TweenMax.to('.box1_1', 0, {left: "0%"});
}
